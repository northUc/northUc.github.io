# 配合express搭建

  let express = require('express')
    let Mock = require('mockjs')
    let app = express()
    app.get('/user',function(req,res){
    let result = Mock.mock({
      code:0,
      message:'成功',
      "data|20":[{
          "name":"@cname",
          "userId":"@id",
          "createAt":"@datetime"
        }]
      })
      res.json('自动生成随机数据',result)
    })
    app.listen(7000)
